#version 150
uniform sampler2D tex;

in vec2 v_texcoord;

out vec4 color;

void main(){
    vec2 texsize = 1.0/textureSize(tex,0);
    vec4 total=vec4(0,0,0,0);
    for(int dy=-1;dy<=1;dy++){
        for(int dx=-1;dx<=1;dx++){
            total += texture(tex,(v_texcoord + vec2(dx,dy)*texsize));
        }
    }
    color = total/9.0;
    color.a=1.0;
}
